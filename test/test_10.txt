//Programa que en el que hay dos funciones una a la que se pasan los parámetros por valor y por referencia. Pensado para ver que el paso de parámetros se realiza bien.

    vars {
    var integer x;
    var integer y;
  }

	subprograms {
    subprogram: aumentar_var_valor(integer var){

            var = var + 1;
        }
    }

    subprogram: aumentar_var_referencia(integer *  var){ 
        instructions {
            var = var + 1; 
        }
    }

  }
    
	instructions {

        x = 1;

     call aumentar_var_referencia(var = x);

	}
