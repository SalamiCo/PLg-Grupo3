program: test {
	consts {
		const natural SIZE = 8;
	}

	vars {
		var natural i;
		var natural[SIZE] arr;
	}
	
	subprograms {
		subprogram: fun(integer* i, integer[SIZE]* arr) {
			instructions {
				i = 0;
				while i < SIZE do
					arr[i] = i;
					i = i + 1;
				endwhile;
			}
		}
	}
	
	instructions {
		out('$');
		call fun(i=i, arr=arr);
	}
}
