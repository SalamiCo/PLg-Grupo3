program: test {
	consts {
		const natural SIZE = 8;
	}

	vars {
		var natural res;
	}
	
	subprograms {
		subprogram: fact(natural n, natural* r) {
			instructions {
				out(n);
				if n == 0 then
					r = 1;
				else
					call fact(n=n-1, r=r);
				endif;
			}
		}
	}
	
	instructions {
		call fact(n=6, r=res);
		out(res);
	}
}
