@Programa simple para probar las llamadas a subprogramas. 

program: test9{

    consts {
        const integer x = 10;
    } 
   
    vars {
        var float y;
        var character z;
        var integer y1;
        var integer z1;
    }

	subprograms {
        subprogram: outvariable(integer c){ 
            instructions {
                out(c);
            }
        }

    }
    
	instructions {
     
        y = 15.01;
        z = 'a';
        y1 = (int) y;
        z1 = (int) z;

        call outvariable(c = x);
        call outvariable(c = y);
        call outvariable(c = z);


	}
}
