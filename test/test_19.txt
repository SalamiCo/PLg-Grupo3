@ Programa de ejemplo
program: ejemplo{
	consts{
		const float razon = 166.386;
	}
	vars{
		var integer cantidad;
		var float euros;
		var float centimos;
		var float pesetas;		
	}
	instructions{
		@in(euros);
		euros = 3;
		cantidad = (int)euros;
		centimos = euros - cantidad;
		out(cantidad);
		out ('.');
		out(' ');
		out (centimos);
		@in(euros);
		euros = 4;
		swap2();
		pesetas = euros / razon;
		out (pesetas);
	}
}
